Naloge

1


(a) Vstavi manjkajoči del, da bo poizvedba vrnila vse vrednosti v tabeli stranka. 

SELECT * FROM stranka;

(b) Zapiši poizvedbo, ki bo vrnila vrednosti stolpca kraj v tabeli stranka. 

SELECT kraj FROM stranka;

(c) Izberi vse različne vrednosti iz stolpca drzava v tabeli stranka. 

SELECT DISTINCT drzava FROM stranka;

2


(a) Izberi vse zapise iz tabele stranka, pri katerih ima stolpec kraj vrednost 'Maribor'. 

SELECT * FROM stranka WHERE kraj = 'Maribor';

(b) Izberi vse zapise iz tabele stranka, pri katerih ima stolpec stranka_id vrednost 32. 

SELECT * FROM stranka WHERE stranka_id = 32;

(c) Izberi vse zapise iz tabele stranka, pri katerih ima stolpec kraj vrednost 'Ljubljana' in stolpec posta vrednost 1108. 

SELECT * FROM stranka WHERE kraj = 'Ljubljana' AND posta = 1108;

(d) Izberi vse zapise iz tabele stranka, pri katerih ima stolpec kraj vrednost 'Lendava' ali 'Piran'. 

SELECT * FROM stranka WHERE kraj = 'Lendava' OR kraj = 'Piran';


3


(a) Izberi vse zapise iz tabele stranka in jih uredi naraščajoče glede na stolpec kraj. 

SELECT * FROM stranka ORDER BY kraj;

(b) Izberi vse zapise iz tabele stranka in jih uredi padajoče glede na stolpec kraj. 

SELECT * FROM stranka ORDER BY kraj DESC;

(c) Izberi vse zapise iz tabele stranka in jih uredi po abecedi naraščajoče najprej glede na stolpec drzava, nato glede na stolpec kraj. 

SELECT * FROM stranka ORDER BY drzava, kraj;

4


Vstavi nov zapis v tabelo stranka. 

INSERT INTO stranka ( ime, naslov, kraj, posta, drzava )
VALUES ( 'Janez Novak', 'Mala pot 3', 'Šmarca', 1241, 'Slovenija');

5


(a) Izberi vse zapise iz tabele stranka, pri katerih je stolpec posta prazen. 

SELECT * FROM stranka WHERE posta IS NULL;

(a) Izberi vse zapise iz tabele stranka, pri katerih stolpec posta ni prazen. 

SELECT * FROM stranka WHERE posta IS NOT NULL;
