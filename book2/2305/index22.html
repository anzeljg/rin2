Naloge

6


(a) Izbriši zapise iz tabele stranka, pri katerih ima stolpec drzava vrednost 'Italija'. 

DELETE FROM stranka WHERE drzava = 'Italija';

(b) Izbriši zapise iz tabele stranka, pri katerih ima stolpec kraj vrednost 'Milano'. 

DELETE FROM stranka WHERE kraj = 'Milano';

(c) Izbriši vse zapise iz tabele stranka. 

DELETE FROM stranka;

7


(a) Uporabi funkcijo MIN in izberi zapis z najmanjšo vrednostjo v stolpcu cena. 

SELECT MIN(cena) FROM izdelek;

(b) Uporabi funkcijo MAX in izberi zapis z največjo vrednostjo v stolpcu cena. 

SELECT MAX(cena) FROM izdelek;

(c) Uporabi ustrezno funkcijo, ki vrne število zapisov, pri katerih je cena enaka 18. 

SELECT COUNT(*) FROM izdelek WHERE cena = 18;

(d) Uporabi ustrezno funkcijo za izračun povprečne cene vseh izdelkov. 

SELECT AVG(cena) FROM izdelek;

(e) Uporabi ustrezno funkcijo za izračun vsote cen vseh izdelkov. 

SELECT SUM(cena) FROM izdelek;


8


(a) Posodobi stolpec kraj pri vseh zapisih v tabeli stranka. 

UPDATE stranka SET kraj = 'Novo mesto';

(b) Posodobi vrednost stolpca kraj na 'Postojna', vendar le pri zapisih v tabeli stranka, ki imajo vrednost stolpca drzava enako 'Slovenija'. 

UPDATE stranka SET kraj = 'Postojna' WHERE drzava = 'Slovenija';

(c) Posodobi vrednosti stolpcev kraj in drzava. 

UPDATE stranka SET kraj = 'Oslo' , drzava = 'Norveška'
WHERE stranka_id = 32;

9


(a) Izberi vse zapise, pri katerih se vrednost stolpca kraj začne s črko »a«. 

SELECT * FROM stranka WHERE kraj LIKE 'a%';

(b) Izberi vse zapise, pri katerih se vrednost stolpca kraj konča s črko »a«. 

SELECT * FROM stranka WHERE kraj LIKE '%a';

(c) Izberi vse zapise, pri katerih vrednost stolpca kraj vsebuje črko »a«. 

SELECT * FROM stranka WHERE kraj LIKE '%a%';

(d) Izberi vse zapise, pri katerih se vrednost stolpca kraj začne z »a« in konča z »b«. 

SELECT * FROM stranka WHERE kraj LIKE 'a%b';

(e) Izberi vse zapise, pri katerih se vrednost stolpca kraj ne začne s črko »a«. 

SELECT * FROM stranka WHERE kraj NOT LIKE 'a%';
