Igra Black Jack

Rokovalnik izpisa na platno izpiše ustrezno besedilo. Za izris kart pa pokliče metodo izpis razreda Igralec, ki poskrbi za izpis vseh kart, ki jih ima igralec v roki. Metoda Igralec.izpis() pa nadalje pokliče metodo izpis razreda Karta, ki poskrbi za izpis oziroma izris slike karte na ustreznem mestu na platnu.

def izpis(platno):
  global izid, igralec, delivec
  igralec.izpis(platno, [50, 220])
  delivec.izpis(platno, [50, 400], True)

  txt1 = "Black Jack"
  txt2 = "Tvoje karte"
  txt3 = "Jackove karte"
  txt4 = "STANJE: " + izid
  txt5 = "REZULTAT: zmage: " + str(REZULTAT["ZMAGE"]) \
         + ", skupaj: " + str(REZULTAT["SKUPAJ"])

  platno.draw_text(txt1, [50, 70], 40, "white")
  platno.draw_text(txt2, [50, 200], 20, "white")
  platno.draw_text(txt3, [50, 380], 20, "white")
  platno.draw_text(txt4, [50, 140], 20, "white")
  platno.draw_text(txt5, [50, 560], 20, "white")

Razrede Karta, Igralec in Komplet smo sicer spoznali že v prejšnji učni enoti. Tukaj smo jih samo še priredili, da jih lahko uporabimo v GUI vmesniku programa.

  9 igra_poteka = False

 13 # Definicija globalnih spremenljivk za karte
 14 BARVE = ("C", "S", "H", "D") # Club/Spade/Heart/Diamond
 15 OPISI = ("A", "2", "3", "4", "5", "6", "7", "8", "9",
 16          "X", "J", "Q", "K")
 17 VREDNOSTI = {"A":1, "2":2, "3":3, "4":4, "5":5,
 18              "6":6, "7":7, "8":8, "9":9, "X":10,
 19              "J":10, "Q":10, "K":10}

 22 # Definicija razreda 'Karta'
 23 class Karta:
 24 
 25   # Ustvari objekt 'Karta'
 26   def __init__(self, barva, opis):
 27     if (barva in BARVE) and (opis in OPISI):
 28       self._barva = barva
 29       self._opis = opis
 30     else:
 31       self._barva = None
 32       self._opis = None
 33       print("Neveljavna karta: ", barva, opis)
 34 
 35   # Vrne dvo-znakovno predstavitev karte
 36   def __str__(self):
 37     return self._barva + self._opis
 38 
 39   # Vrne barvo karte
 40   def vrni_barvo(self):
 41     return self._barva
 42 
 43   # Vrne opis ali rang karte
 44   def vrni_opis(self):
 45     return self._opis
 46 
 47   # Izriše sliko karte na platnu, na danem položaju
 48   def izpis(self, platno, poz):
 49     k = self._barva + self._opis
 50     platno.create_image(poz[0], poz[1], \
 51       anchor=tk.NW, image=KARTE[k])

 53 
 54 # Definicija razreda 'Komplet'
 55 class Komplet:
 56 
 57   # Ustvari objekt 'Komplet', ki vsebuje vseh 52 kart
 58   def __init__(self):
 59     self._karte = []
 60     for barva in BARVE:
 61       for opis in OPISI:
 62         karta = Karta(barva, opis)
 63         self._karte.append(karta)
 64 
 65   # Premešaj komplet kart
 66   def premesaj(self):
 67     random.shuffle(self._karte)
 68 
 69   # Deli naslednjo karto iz kompleta
 70   def deli_karto(self):
 71     return self._karte.pop()
 72 
 73   # Vrne besedilno predstavitev kompleta kart
 74   def __str__(self):
 75     sporocilo = "Komplet vsebuje"
 76     for karta in self._karte:
 77       sporocilo += " " + str(karta)
 78     return sporocilo
 79 
 80 
 81 # Definicija razreda 'Igralec'
 82 class Igralec(Komplet):
 83 
 84   # Ustvari objekt 'Igralec', ki privzeto ni delivec
 85   def __init__(self, delivec=False):
 86     self._karte = []
 87     self._delivec = delivec
 88 
 89   # Vrne predstavitev kart, ki jih v roki drži igralec
 90   def __str__(self):
 91     sporocilo = "V roki drži"
 92     skrita = False
 93     if self._delivec:
 94       skrita = True
 95     for karta in self._karte:
 96 	  # Skrije prvo karto delivca
 97       if self._delivec and skrita:
 98         sporocilo += " ??"
 99         skrita = False
100       else:
101         sporocilo += " " + str(karta)
102     return sporocilo
103 
104   # Igralcu doda nov objekt 'Karta'
105   def dodaj_karto(self, karta):
106     self._karte.append(karta)
107 
108   # Vrne skupno vrednost kart, ki jih igralec drži v roki
109   def vrednost_kart(self):
110     # Če ima igralec v roki karto as, jo šteje kot 1 in
111     # nato, če vrednost kart v roki ne preseže 21, doda
112     # vrednosti še 10 - saj je as vreden 1 ali 11 točk.
113     vrednost = 0
114     asi = 0
115     for karta in self.karte:
116       vrednost += VREDNOSTI[karta._opis]
117       if karta._opis == "A":
118         asi += 1
119     for a in range(asi):
120       # Če je vrednost kart v roki igralca dovolj nizka,
121       # lahko štejemo asa kot 11, namesto 1...
122       # To pomeni: vrednosti dodaj dodatnih 10 točk!
123       if vrednost <= 11:
124         vrednost += 10
125     return vrednost
126 
127   # Izriše karte - uporabi metodo izpis razreda 'Karta'
128   def izpis(self, platno, poz, delivec=False):
129     ROB = 10
130     i = 0
131     for karta in self._karte:
132       karta.izpis(platno, [poz[0]+i*(VEL[0]+ROB),poz[1]])
133       # Če izrisujemo karte delivca in igra še poteka,
134       # potem skrijemo prvo karto delivca tako, da čez
135       # njo izrišemo sliko hrbtne strani karte.
136       if (delivec and igra_poteka and i == 0):
137         platno.create_image(poz[0], poz[1], \
138           anchor=tk.NW, image=KARTE['BG'])
139       i += 1
140

142 # Določi rokovalnik gumba 'Deli'
143 def deli():
144   # Če kliknemo gumb medtem, ko igra poteka
145   global izid, REZULTAT, igra_poteka
146   if igra_poteka:
147     izid = "Jack je zmagal."
148     REZULTAT["SKUPAJ"] += 1
149 
150   global komplet
151   komplet = Komplet()
152   komplet.premesaj()
153 
154   global igralec, delivec
155   igralec = Igralec()
156   delivec = Igralec(True)
157   # Izmenično razdeli dve karti igralcu in delivcu
158   igralec.dodaj_karto(komplet.deli_karto())
159   delivec.dodaj_karto(komplet.deli_karto())
160   igralec.dodaj_karto(komplet.deli_karto())
161   delivec.dodaj_karto(komplet.deli_karto())
162   print("Igralec: " + str(igralec))
163   print("Delivec: " + str(delivec))
164 
165   izid = "Na vrsti si. Vleci ali obdrži?"
166   igra_poteka = True
167   izpis()
168 
169 
170 # Določi rokovalnik gumba 'Vleci'
171 def vleci():
172   # Če igra poteka, potem igralec vleče karto
173   global izid, REZULTAT, igra_poteka, komplet, igralec
174   if igra_poteka:
175     igralec.dodaj_karto(komplet.deli_karto())
176     print("Vrednost kart igralca: " +
177           str(igralec.vrednost_kart()))
178   # Če igralec preseže 21, potem priredi sporočilo
179   # spremeljivki izid, posodobi vrednosti spremeljivk
180   # igra_poteka in REZULTAT
181   if igralec.vrednost_kart() > 21:
182     izid = "Imaš več kot 21. Jack je zmagal. Nova igra?"
183     igra_poteka = False
184     print(izid)
185     REZULTAT["SKUPAJ"] += 1
186     print(REZULTAT)
187   izpis()
188 
189 
190 # Določi rokovalnik gumba 'Obdrži'
191 def obdrzi():
192   # Če igra poteka, potem delivec vleče karte, dokler
193   # ni njihova skupna vrednost 17 ali več
194   global izid, REZULTAT, igra_poteka, komplet, delivec
195   while delivec.vrednost_kart() < 17:
196     delivec.dodaj_karto(komplet.deli_karto())
197     print("Vrednost kart delivca: " +
198           str(delivec.vrednost_kart()))
199   igra_poteka = False
200   # Priredi sporočilo spremeljivki izid, posodobi
201   # vrednosti spremeljivk igra_poteka in REZULTAT
202   if delivec.vrednost_kart() > 21:
203     izid = "Jack ima več kot 21. Zmagal si! Nova igra?"
204     print(izid)
205     REZULTAT["SKUPAJ"] += 1
206     REZULTAT["ZMAGE"] += 1
207     print(REZULTAT)
208   else:
209     # Določi zmagovalca
210     REZULTAT["SKUPAJ"] += 1
211     if igralec.vrednost_kart() > delivec.vrednost_kart():
212       izid = "Zmagal si! Nova igra?"
213       REZULTAT["ZMAGE"] += 1
214     else:
215       izid = "Jack je zmagal. Nova igra?"
216     print(izid)
217     print(REZULTAT)
218   izpis()
