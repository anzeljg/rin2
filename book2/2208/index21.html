<!DOCTYPE html>
<html>
  <head>
    <title>Zabave</title>
    <meta http-equiv="content-type" content="text/html; charset=UTF-8">
    <style>
      .prosojno {background-color:rgba(255,255,255,0.8)}
      .prosojno a {color:maroon}
    </style>
  </head>
  <body background="ozadje.png">
    <table border="3" cellpadding="2" cellspacing="2">
      <tr align="center" bgcolor="pink">
        <td colspan="2">
          <big><b>Otroške rojstnodnevne zabave</b></big>
        </td>
      </tr>
      <tr class="prosojno">
        <td width="50%">
          <b>Izbirate lahko med:</b>
          <ul>
            <li>športno zabavo ali</li>
            <li>ustvarjalno delavnico</li>
            <li>v obeh primerih pa ponudba obsega še:</li>
            <ul>
              <li>triurno animacijo</li>
              <li>pico in sok</li>
            </ul>
          </ul>
        </td>
        <td width="50%" rowspan="2" align="center">

      </tr>
    </table>
  </body>
</html>
